Setup

# load packages
library(stringr)
library(tidyverse)
library(lubridate)
source("../my_R_funs.R")
source("../my_NA_strings.R")
# load data
dat <- read_csv("../data/library questionnaires RAW 2024-06-24.csv", na = na_strings, show_col_types = F) |> tail(-5)

Cleaning

# rename demog vars
dat <- dat |> 
  rename(id = "Q1",
         age = "Q2",
         gender = "Q3",
         gender_describe = "Q3_4_TEXT",
         autism = "Q4",
         adhd = "Q5",
         dyslexia = "Q6",
         migraine = "Q7",
         migraine_freq = "Q147",
         )

# rename scale vars
dat <- dat |> 
  rename_at(vars(starts_with("VDS")), ~paste0("vds_", seq_along(.))) |> 
  rename_at(vars(starts_with("AQ")), ~paste0("aq_", seq_along(.))) |> 
  rename_at(vars(starts_with("Q")), ~paste0("gsq_", seq_along(.))) |> 
  rename_at(vars(starts_with("SRS")), ~paste0("srs_", seq_along(.)))
# change data types
dat <- dat |> mutate(across(c(age, vds_1:srs_65), as.numeric))

# recode factor levels
dat <- dat |> 
  mutate(gender = fct_recode(gender, "man" = "1", "non-binary" = "2", "woman" = "3", "self-describe" = "4", "PNTS" = "5"),
         migraine = fct_recode(migraine, "yes" = "1", "no" = "2")) |> 
  mutate_at(vars(autism:dyslexia), ~fct_recode(.x, "clinical" = "1", "self" = "2", "non" = "3"))
# select relevant vars |> remove under 18s |> recode birth year to age |> drop NAs
dat <- dat |> 
  select(id:srs_65) |> 
  filter(age >= 18) |> 
  mutate(age = ifelse(str_length(age) == 4, year(Sys.Date()) - age, age)) |> 
  drop_na(vds_1)

GSQ

  • 42 items scored on a 5-point scale (0-4) – none reverse-scored

  • Total score range: 0-168

  • Two subscale axes

    • Hyper- and hypo-sensitivity

    • Modality: visual, auditory, gustatory, olfactory, tactile, vestibular, proprioceptive

# load subscale codes
gsq_subcodes <- read_csv("../data/GSQ subscale codes.csv", show_col_types = F)

# total scores
dat <- dat |> mutate(gsq_total = rowSums(dat |> select(gsq_1:gsq_42)))

# hyper/hypo scores
dat <- dat |> 
  mutate(gsq_hyper = eval(parse(text = paste0(gsq_subcodes$item[gsq_subcodes$hyper_hypo == "hyper"], collapse = " + "))),
         gsq_hypo = eval(parse(text = paste0(gsq_subcodes$item[gsq_subcodes$hyper_hypo == "hypo"], collapse = " + ")))
         )

# modality scores
dat <- dat |>

# source code for eval_subscore()

  parse(text = paste0(item_var[subscale_var == subscale], collapse = " + ")))

SRS

  • 65 items scored on a 4-point scale (0-3) – 15 reverse-scored

  • Raw total score range: 0-195

    • Scores > 153 coded as 153

  • Raw scores converted into T-scores (36-90)

  • Subscales: social awareness, social cognition, social communication, social motivation, RRB

# total & subscale scores
dat <- dat |>
